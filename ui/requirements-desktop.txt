# Optional desktop-app extras for the Market AI Lab dashboard.
#
# Install ON TOP of ui/requirements.txt to run the Dash board as a TRUE native
# desktop application (native OS window + system tray + 24/7 engine
# supervision — see ui/desktop.py), and to package it into a single Windows
# .exe with PyInstaller (see ops/build_exe.bat and ui/MarketAILab.spec):
#
#   pip install -r ui/requirements.txt -r ui/requirements-desktop.txt
#   python ui/desktop.py            # run from source
#   ops\build_exe.bat               # build dist\MarketAILab.exe (Windows/MSVC)
#
# pywebview uses the platform's native webview (WebView2/EdgeChromium on
# Windows — built into Windows 10/11; WebKit/GTK on Linux, where you may also
# need system packages e.g. python3-gi, gir1.2-webkit2-4.1, or the Qt backend).
#
# EXACT version pins (Task 9 security hardening): reproducible installs, no
# silent supply-chain drift. Bump deliberately after review.
pywebview==5.3.2
pystray==0.19.5
pillow==10.4.0
pyinstaller==6.11.1

# --- Linux backend note ---------------------------------------------------- #
# pywebview needs a native webview backend. Two options on Linux:
#   * GTK  (system pkgs: python3-gi gir1.2-webkit2-4.1) - default on older setups
#   * Qt   (pip pkgs below)                             - more reliable on
#                                                          GTK4 / Python 3.13+
# If the GTK window fails to open (e.g. a GTK4/Python-3.14 'initialized' error),
# install the Qt backend and ops/run_desktop.sh will prefer it automatically:
#   pip install pyqt6 pyqt6-webengine qtpy
# (uncomment the next three lines to make Qt part of the default desktop install)
# pyqt6==6.7.1
# pyqt6-webengine==6.7.0
# qtpy==2.4.1
